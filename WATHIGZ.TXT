# ==========================================
# Mathematical Logic & Calculations (Equations)
# ==========================================
The system calculates critical electrical parameters using digital signal processing.

1. True RMS Voltage (V_rms)
The system removes the DC offset (V_mean) and calculates True RMS using 256 samples:
$$V_{rms} = \sqrt{\frac{1}{N} \sum_{n=0}^{N-1} (v[n] - V_{mean})^2} \times Scale_V$$

2. True RMS Current (I_rms)
Calculated similarly, but filtered using a low-pass exponential moving average to remove low-load noise:
$$I_{raw} = \sqrt{\frac{1}{N} \sum_{n=0}^{N-1} (i[n] - I_{mean})^2} \times Scale_I$$
$$I_{filtered} = (I_{old} \times 0.8) + (I_{new} \times 0.2)$$

3. Active Power (P) and Power Factor (PF)
Active power is the average of instantaneous voltage multiplied by instantaneous current:
$$P = \frac{1}{N} \sum_{n=0}^{N-1} (v[n] \times i[n]) \times Scale_V \times Scale_I$$
Power Factor is the ratio of Active Power to Apparent Power:
$$PF = \frac{P}{V_{rms} \times I_{rms}}$$

4. Total Harmonic Distortion (THD)
Calculated using Fast Fourier Transform (FFT). It finds the fundamental frequency (V_1) and sums the squared magnitudes of harmonics (V_h):
$$THD = \frac{\sqrt{\sum_{h=2}^{k} V_h^2}}{V_1}$$

# 2. دوال معالجة الإشارات (Signal Processing Functions)
```cpp
double getMag(double *v, int bin) {
    if (bin <= 0 || bin >= SAMPLES / 2) return 0;
    return v[bin];
}

int findFund(double *v) {
    int start = (40.0 * SAMPLES) / SAMPLING_FREQ;
    int end   = (70.0 * SAMPLES) / SAMPLING_FREQ;
    double maxV = 0;
    int best = start;
    for (int i = start; i <= end; i++) {
        double m = getMag(v, i);
        if (m > maxV) {
            maxV = m;
            best = i;
        }
    }
    return best;
}

double calculateTHD(double *v, int fund) {
    double f = getMag(v, fund);
    if (f < 1e-9) return 0;
    double sumSq = 0;
    for (int h = 2; h * fund < SAMPLES / 2; h++) {
        double m = getMag(v, h * fund);
        if (m < 0.01 * f) continue;
        sumSq += m * m;
    }
    return sqrt(sumSq) / f;
}

double estimateFrequencyZeroCross(double *v) {
    double firstCross = -1;
    double secondCross = -1;
    for (int i = 1; i < SAMPLES; i++) {
        if (v[i - 1] < 0 && v[i] >= 0) {
            double denom = (v[i] - v[i - 1]);
            if (fabs(denom) < 1e-12) continue;
            double frac = (0.0 - v[i - 1]) / denom;
            double crossing = (i - 1) + frac;
            if (firstCross < 0)
                firstCross = crossing;
            else {
                secondCross = crossing;
                break;
            }
        }
    }
    if (firstCross >= 0 && secondCross > firstCross) {
        double periodSamples = secondCross - firstCross;
        return SAMPLING_FREQ / periodSamples;
    }
    return 0;
}
```

void calibrateZeroI() {
    tft.fillScreen(TFT_BLACK);
    tft.setCursor(10, 100);
    tft.setTextColor(TFT_WHITE, TFT_BLACK);
    tft.setTextSize(2);
    tft.print("Calibrating Current...");
    float sum = 0;
    for (int j = 0; j < 50; j++) {
        float max_v = 0, min_v = 3.3;
        uint32_t start = millis();
        while (millis() - start < 20) {
            float v = analogReadMilliVolts(CURRENT_PIN) / 1000.0;
            if (v > max_v) max_v = v;
            if (v < min_v) min_v = v;
        }
        sum += (max_v - min_v) * scaleFactorI;
        delay(10);
    }
    zeroOffsetI = sum / 50.0;
}

void setup() {
    Serial.begin(115200);
    tft.init();
    tft.setRotation(1);
    tft.fillScreen(TFT_BLACK);
#if defined(ESP32)
    analogReadResolution(12);
    analogSetAttenuation(ADC_11db);
#endif
    calibrateZeroI();
    for (int i = 0; i < windowSize; i++)
        readingsV[i] = 0;
}

# 4. أخذ العينات وحساب الطاقة (Sampling & Power Calculations)
```cpp
void loop() {
    unsigned long t = micros();
    // جمع العينات
    for (int i = 0; i < SAMPLES; i++) {
        while ((long)(micros() - t) < 0);
        float vRaw = analogReadMilliVolts(VOLTAGE_PIN) / 1000.0;
        float iRaw = analogReadMilliVolts(CURRENT_PIN) / 1000.0;
        vReal_V[i] = vRaw;
        vReal_I[i] = iRaw;
        vImag_V[i] = 0;
        vImag_I[i] = 0;
        t += SAMPLE_PERIOD_US;
    }

    // إزالة DC + حساب RMS الحقيقي
    double meanV = 0, meanI = 0;
    for (int i = 0; i < SAMPLES; i++) {
        meanV += vReal_V[i];
        meanI += vReal_I[i];
    }
    meanV /= SAMPLES;
    meanI /= SAMPLES;
    double sumV2 = 0, sumI2 = 0, sumP = 0;
    for (int i = 0; i < SAMPLES; i++) {
        double v = vReal_V[i] - meanV;
        double c = vReal_I[i] - meanI;
        vReal_V[i] = v;
        vReal_I[i] = c;
        sumV2 += v * v;
        sumI2 += c * c;
        sumP  += v * c;
    }

    // الجهد الحقيقي
    float instantV = sqrt(sumV2 / SAMPLES) * scaleFactorV;
    totalV -= readingsV[readIndexV];
    readingsV[readIndexV] = instantV;
    totalV += readingsV[readIndexV];
    readIndexV++;
    if (readIndexV >= windowSize) readIndexV = 0;
    float averageVrms = totalV / windowSize;

    // التيار الحقيقي
    float currentRaw = sqrt(sumI2 / SAMPLES) * scaleFactorI;
    currentRaw -= zeroOffsetI;
    if (currentRaw < 0) currentRaw = 0;
    filteredCurrentI = (filteredCurrentI * 0.8) + (currentRaw * 0.2);

    // القدرة و PF
    double P = (sumP / SAMPLES) * scaleFactorV * scaleFactorI;
    if (P < 0) P = 0;
    double PF = 0;
    if (averageVrms > 0.01 && filteredCurrentI > 0.01)
        PF = P / (averageVrms * filteredCurrentI);
    if (PF > 1) PF = 1;
    if (PF < -1) PF = -1;
```
